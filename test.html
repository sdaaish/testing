Function refreshenv {
    $paths = @(
        ([System.Environment]::GetEnvironmentVariable("Path","Machine") -split ";")
        ([System.Environment]::GetEnvironmentVariable("Path","User") -split ";")
    )
    $env:path = ($paths|Select-Object -Unique) -join ";"
}
